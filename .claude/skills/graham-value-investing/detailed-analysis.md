fn calculate_dcf(
    free_cash_flow: f64,
    growth_rate: f64,
    discount_rate: f64,
    terminal_growth: f64,
    years: u32,
) -> f64 {
    let mut pv = 0.0;
    let mut fcff = free_cash_flow;

    for year in 1..=years {
        fcff = fcff * (1.0 + growth_rate);
        let discount_factor = (1.0 + discount_rate).powi(year as i32);
        pv += fcff / discount_factor;
    }

    // Terminal Value
    let terminal_value = fcff * (1.0 + terminal_growth)
        / (discount_rate - terminal_growth);
    let terminal_discount = (1.0 + discount_rate).powi(years as i32);
    pv += terminal_value / terminal_discount;

    pv
}
